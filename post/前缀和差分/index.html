一维前缀和的应用

输入一个长度为 n 的整数序列。

接下来再输入 m个询问，每个询问输入一对 l,r。

对于每个询问，输出原序列中从第 l个数到第 r个数的和。

输入格式

第二行包含 n个整数，表示整数数列。

接下来 m 行，每行包含两个整数 l 和 r，表示一个询问的区间范围。

输出格式

共 m 行，每行输出一个询问的结果。

#include <iostream>

using namespace std;

const int N = 100010;

int n, m;
int a[N], s[N];

int main()
{
	scanf("%d%d",&n, &m);
	// 读入a[N]，N个数据
	for (int i = 1; i <= n; i++ ) scanf("%d", &a[i]);
	// 计算前缀和数组
	for (int i = 1; i <= n; i++) s[i] = s[i - 1] + a[i];
	// m次询问
	while (m --)
	{
		int l, r;
		scanf("%d%d", &l, &r);
		// 使用公式直接计算
		printf("%d\n",s[r] - s[l - 1]);
	}	
	system("pause");
	return 0;
}

二维前缀和的应用

#include <iostream>

using namespace std;

const int N = 1010;

int n, m, q;
int a[N][N], s[N][N];

int main()
{
	scanf("%d%d%d",&n, &m, &q);
	// 读入a[N][N]
	for (int i = 1; i <= n; i++ )
		for (int j = 1; j <= m; j++)
			scanf("%d", &a[i][j]);
	// 计算前缀和数组
	for (int i = 1; i <= n; i++) 
		for (int j = 1; j <= m; j++)
			s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + a[i][j];
	// q次询问
	while (q --)
	{
		int x1, y1, x2, y2;
		scanf("%d%d%d%d",&x1, &y1, &x2, &y2);
		printf("%d\n",s[x2][y2] - s[x2][y1 - 1] - s[x1 - 1][y2] + s[x1 -1][y1 - 1]);
	}	
	return 0;
}

#include <iostream>

using namespace std;

const int N = 100010;

int n, m;
int a[N], b[N];

void insert(int l, int r, int c)
{
	b[l] += c;
	b[r + 1] -= c;
}

int main()
{
	scanf("%d%d",&n, &m);
	// 读入a[N]
	for (int i = 1; i <= n; i++ ) scanf("%d", &a[i]);
	// 把a数组中的每个数插入到b数组中
	for (int i = 1; i <= n; i++) insert(i,i,a[i]);
	// m次询问
	while (m --)
	{
		int l, r, c;
		scanf("%d%d%d", &l, &r, &c);
		insert(l, r, c);
	}
	// 计算出a[i]所对应的每个值，用b[i]来进行计算
	for (int i = 1; i <= n; i++) b[i] += b[i-1];
	// 输出最后的结果
	for (int i = 1; i <= n; i++) printf("%d ", b[i]);
	return 0;
}

二维差分的应用

输入一个 nn 行 m 列的整数矩阵，再输入 q 个操作，每个操作包含五个整数 x1,y1,x2,y2,c，其中 (x1,y1) 和 (x2,y2) 表示一个子矩阵的左上角坐标和右下角坐标。

#include <iostream>

using namespace std;

const int N = 1010;

int n, m, q;
int a[N][N], b[N][N];

void insert(int x1, int y1, int x2, int y2, int c)
{
	b[x1][y1] += c;
	b[x2 + 1][y1] -= c;
	b[x1][y2 + 1] -= c;
	b[x2 + 1][y2 + 1] += c;
}

int main()
{
	scanf("%d%d%d",&n, &m, &q);
	// 读入a[N][N]
	for (int i = 1; i <= n; i++ )
		for (int j = 1; j <= m; j++)
			scanf("%d", &a[i][j]);
	// 把a矩阵中的每个数插入到b矩阵中
	for (int i = 1; i <= n; i++) 
		for (int j = 1; j <= m; j++)
			insert(i, j, i, j, a[i][j]);
	// q次询问
	while (q --)
	{
		int x1, y1, x2, y2, c;
		scanf("%d%d%d%d%d",&x1, &y1, &x2, &y2, &c);
		insert(x1, y1, x2, y2, c);
	}	
	// 求前缀和（用b来表示a）
	for (int i = 1; i <= n; i++) 
		for (int j = 1; j <= m; j++)
			b[i][j] = b[i - 1][j] +  b[i][j - 1] - b[i - 1][j - 1];
	// 输出结果
	for (int i = 1; i <= n; i++) 
	{
		for (int j = 1; j <= m; j++) printf("%d ", b[i][j]);
		printf("\n");
	}
	system("pause");
	return 0;
}
